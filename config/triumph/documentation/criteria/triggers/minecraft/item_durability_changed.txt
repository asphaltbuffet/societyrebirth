These are the functions which can be called on a minecraft:item_durability_changed 
criteria trigger.



setDurabilityMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min item durability.

setDeltaMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max delta.

setDeltaMin:
    Arguments:
        float
    Usage:
        min
    Notes:
        Sets the min delta.

setDeltaMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max delta.

setDurabilityMax:
    Arguments:
        float
    Usage:
        max
    Notes:
        Sets the max item durability.

setItem:
    Arguments:
        ItemBlockData
    Usage:
        item data
    Notes:
        Sets the required item.

setDurabilityMinMax:
    Arguments:
        float, float
    Usage:
        min, max
    Notes:
        Sets the min and max item durability.
