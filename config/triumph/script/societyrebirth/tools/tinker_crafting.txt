//Set the icon item (stencil table)
setIcon(<tconstruct:tooltables, meta:1>)

//Set the title String
setTitle("Parts to make Parts")

//Set the description String
setDescription("Craft a tinker's stencil table, part builder, and tool station.")

//Set the parent advancement. The syntax is "modid:folder/path/to/the/advancement/then/filename"
setParent("triumph:societyrebirth/tools/crafting_table")
setRequiresParent(true)

//Set the position
//setPos(12, 12)

//Set this to use direct lines, so I can have a pretty shape
drawDirectLines(true)

//Adds criteria named "crafted" with the trigger type "triumph:player_crafted_item". This function returns the criteria as an object
criteria = addCriteria("crafted_cts", "triumph:player_crafted_item")

//Sets the item for the criteria. This is the item that needs to be crafted.
// stencil table
criteria.setItem(<tconstruct:tooltables, meta:1>)
// part builder
criteria.setItem(<tconstruct:tooltables, meta:2>)
// tool station
criteria.setItem(<tconstruct:tooltables, meta:3>)

// Require all to be crafted
setRequirements("all")
